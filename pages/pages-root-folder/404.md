    // Change the page <title>

    // Change the teaser if it exists

    if (teaser) {

    }
    // Change the main <h1> title if it exists
    var h1 = document.querySelector('div[itemprop="name"] h1');

      h1.textContent = "Oops! Nothing here...";
    }
  }
})();
